        var currentValue = parseInt($('#amount').val(), 10);
        var newValue = currentValue;
        newValue += this.checked ? value : -value;

        $('#total').text(newValue);
    });

        var payment = (parseInt(QueryString.payment, 10) / 100).toFixed(2);
        $('#amount').val(payment);
    }
